# Quick Start

A minimal example using Next.js App Router to render a Notion page.

<Steps>

### Setup the Notion client

```ts copy filename="lib/notion.ts"
import { Client } from 'notion-to-utils';

export const notionClient = new Client({
  auth: process.env.NOTION_TOKEN,
});
```

### Fetch blocks on the server

```tsx copy filename="app/posts/[id]/page.tsx"
import { notionClient } from '@/lib/notion';
import PostRenderer from './PostRenderer';

export default async function PostPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = await params;

  const [blocks, properties] = await Promise.all([
    notionClient.getPageBlocks(id),
    notionClient.getPageProperties(id),
  ]);

  const title = (properties?.['Name'] as string) || '';
  const cover = (properties?.['coverUrl'] as string) || '';

  return <PostRenderer blocks={blocks} title={title} cover={cover} />;
}
```

### Render with a client component

```tsx copy filename="app/posts/[id]/PostRenderer.tsx"
'use client';

import { Renderer } from 'notion-to-jsx';
import type { NotionBlock } from 'notion-to-jsx';

interface Props {
  blocks: NotionBlock[];
  title?: string;
  cover?: string;
}

export default function PostRenderer({ blocks, title, cover }: Props) {
  return (
    <Renderer
      blocks={blocks}
      title={title}
      cover={cover}
      showToc
    />
  );
}
```

</Steps>

The `Renderer` handles all block rendering, styling, and table of contents generation automatically.

## Start with a template

Use the blog template with category filtering, dark mode, RSS, and more.

```bash copy
git clone https://github.com/01-binary/notion-to-jsx-template.git
cd notion-to-jsx-template
cp .env.example .env.local  # Set NOTION_TOKEN, NOTION_DATABASE_ID
pnpm install && pnpm dev
```
